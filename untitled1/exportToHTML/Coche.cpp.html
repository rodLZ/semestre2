#include <iostream>
#include <string>
#include "Coche.h"

Coche::Coche(int newValor) : svalor(newValor) {}

Coche::Coche(const Coche &coche) : svalor(coche.svalor) {}

Coche::Coche(Coche &&coche) noexcept : svalor(coche.svalor) {} // Implementado aqu√≠

Coche &Coche::operator=(const Coche &coche) {
    if (this != &coche) {
        svalor = coche.svalor;
    }
    return *this;
}

Coche &Coche::operator=(Coche &&coche) noexcept {
    if (this != &coche) {
        svalor = coche.svalor;
        coche.svalor = 0;
    }
    return *this;
}

Coche &Coche::operator=(int valor) {
    svalor = valor;
    return *this;
}

Coche &Coche::operator+(const Coche &coche) {
    svalor += coche.svalor;
    return *this;
}

Coche &Coche::operator+(int valor) {
    svalor += valor;
    return *this;
}

Coche &Coche::operator-(const Coche &coche) {
    svalor -= coche.svalor;
    return *this;
}

Coche &Coche::operator-(int valor) {
    svalor -= valor;
    return *this;
}

int Coche::getValor() const {
    return svalor;
}

void Coche::setValor(int newValor) {
    svalor = newValor;
}
